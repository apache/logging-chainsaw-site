260     public void loadLogPanelColorizers() {
261         if (logPanelColorizersModel.getIndexOf(noTab) == -1) {
262             logPanelColorizersModel.addElement(noTab);
263         }
264         for (Object o : allLogPanelColorizers.entrySet()) {
265             Map.Entry entry = (Map.Entry) o;
266             if (!entry.getValue().equals(currentLogPanelColorizer) && (logPanelColorizersModel.getIndexOf(entry.getKey()) == -1)) {
267                 logPanelColorizersModel.addElement(entry.getKey());
268             }
269         }
270         //update search and alternating colors, since they may have changed from another color panel
271         searchDataVectorEntry.set(0, applicationPreferenceModel.getSearchBackgroundColor());
272         searchDataVectorEntry.set(1, applicationPreferenceModel.getSearchForegroundColor());
273         alternatingColorDataVectorEntry.set(0, applicationPreferenceModel.getAlternatingColorBackgroundColor());
274         alternatingColorDataVectorEntry.set(1, applicationPreferenceModel.getAlternatingColorForegroundColor());
275     }

517     JPanel buildUpDownPanel() {
518         JPanel panel = new JPanel(new BorderLayout());
519         JPanel innerPanel = new JPanel();
520         innerPanel.setLayout(new GridLayout(5, 1));
521 
522         final JButton upButton = new JButton(ChainsawIcons.ICON_UP);
523         upButton.setToolTipText("Move selected rule up");
524 
525         final JButton downButton = new JButton(ChainsawIcons.ICON_DOWN);
526         downButton.setToolTipText("Move selected rule down");
527         upButton.setEnabled(false);
528         downButton.setEnabled(false);
529 
530         rulesTable.getSelectionModel().addListSelectionListener(
531             e -> {
532                 if (!e.getValueIsAdjusting()) {
533                     int index = rulesTable.getSelectionModel().getMaxSelectionIndex();
534 
535                     if (index < 0) {
536                         downButton.setEnabled(false);
537                         upButton.setEnabled(false);
538                     } else if ((index == 0) && (rulesTableModel.getRowCount() == 1)) {
539                         downButton.setEnabled(false);
540                         upButton.setEnabled(false);
541                     } else if ((index == 0) && (rulesTableModel.getRowCount() > 1)) {
542                         downButton.setEnabled(true);
543                         upButton.setEnabled(false);
544                     } else if (index == (rulesTableModel.getRowCount() - 1)) {
545                         downButton.setEnabled(false);
546                         upButton.setEnabled(true);
547                     } else {
548                         downButton.setEnabled(true);
549                         upButton.setEnabled(true);
550                     }
551                 }
552             });
553 
554         JPanel upPanel = new JPanel();
555 
556         upPanel.add(upButton);
557 
558         JPanel downPanel = new JPanel();
559         downPanel.add(downButton);
560 
561         innerPanel.add(new JLabel(""));
562         innerPanel.add(upPanel);
563         innerPanel.add(new JLabel(""));
564         innerPanel.add(downPanel);
565         panel.add(innerPanel, BorderLayout.CENTER);
566 
567         upButton.addActionListener(
568             new AbstractAction() {
569                 public void actionPerformed(ActionEvent evt) {
570                     int index = rulesTable.getSelectionModel().getMaxSelectionIndex();
571                     rulesTableModel.moveRowAtIndexUp(index);
572                 }
573             });
574 
575         downButton.addActionListener(
576             new AbstractAction() {
577                 public void actionPerformed(ActionEvent evt) {
578                     int index = rulesTable.getSelectionModel().getMaxSelectionIndex();
579                     rulesTableModel.moveRowAtIndexDown(index);
580                 }
581             });
582 
583         return panel;
584     }

797     class SelectedIcon implements Icon {
798         private boolean isSelected;
799         private int width = 9;
800         private int height = 18;
801         private int[] xPoints = new int[4];
802         private int[] yPoints = new int[4];
803 
804         public SelectedIcon(boolean isSelected) {
805             this.isSelected = isSelected;
806             xPoints[0] = 0;
807             yPoints[0] = -1;
808             xPoints[1] = 0;
809             yPoints[1] = height;
810             xPoints[2] = width;
811             yPoints[2] = height / 2;
812             xPoints[3] = width;
813             yPoints[3] = (height / 2) - 1;
814         }
815 
816         public int getIconHeight() {
817             return height;
818         }
819 
820         public int getIconWidth() {
821             return width;
822         }
823 
824         public void paintIcon(Component c, Graphics g, int x, int y) {
825             if (isSelected) {
826                 int length = xPoints.length;
827                 int[] newXPoints = new int[length];
828                 int[] newYPoints = new int[length];
829 
830                 for (int i = 0; i < length; i++) {
831                     newXPoints[i] = xPoints[i] + x;
832                     newYPoints[i] = yPoints[i] + y;
833                 }
834 
835                 g.setColor(Color.black);
836 
837                 g.fillPolygon(newXPoints, newYPoints, length);
838             }
839         }

842     private class RulesTableModel extends AbstractTableModel{
843         
844         private List<ColorRule> m_data;
845 
846         RulesTableModel(){
847             m_data = new ArrayList<>();
848             addDefaultRules();
849         }
850 
851         List<ColorRule> rules(){
852             return m_data;
853         }
854 
855         void clear(){
856             m_data.clear();
857             fireTableDataChanged();
858         }
859 
860         void addNewRule(){
861             Rule expressionRule = ExpressionRule.getRule("msg=='sample'");
862             ColorRule newRule =
863                     new ColorRule("msg=='sample'", expressionRule, Color.WHITE, Color.BLACK);
864             m_data.add(newRule);
865             fireTableRowsInserted(m_data.size() - 1, m_data.size() - 1);
866         }
867         
868         void deleteRuleAtIndex(int i){
869             m_data.remove(i);
870             fireTableRowsDeleted(i, i);
871         }
872 
873         void moveRowAtIndexDown(int row){
874             if( row <= m_data.size() - 2 && row >= 0){
875                 Collections.swap(m_data, row, row + 1);
876                 fireTableRowsUpdated(row, row + 1);
877             }
878         }
879 
880         void moveRowAtIndexUp(int row){
881             if( row <= m_data.size() && row > 0 ){
882                 Collections.swap(m_data, row, row - 1);
883                 fireTableRowsUpdated(row - 1, row );
884             }
885         }
886 
887         void addDefaultRules(){
888             Map map = colorizer.getRules();
889             for (Object o1 : map.entrySet()) {
890                 Map.Entry entry = (Map.Entry) o1;
891                 logger.debug( "entry: {}", o1 );
892                 //update ruleset list
893                 if (entry.getKey().equals(currentRuleSet)) {
894                     m_data.addAll( (List<ColorRule>)entry.getValue() );
895                 }
896             }
897         }
898 
899         @Override
900         public void setValueAt(Object aValue, int rowIndex, int columnIndex) {
901             if( rowIndex >= m_data.size() ){
902                 return;
903             }
904             String expression = m_data.get(rowIndex).getExpression();
905             Rule rule = m_data.get(rowIndex).getRule();
906             Color backgroundColor = m_data.get(rowIndex).getBackgroundColor();
907             Color foregroundColor = m_data.get(rowIndex).getForegroundColor();
908 
909             switch( columnIndex ){
910                 case 0:
911                     expression = aValue.toString();
912                     rule = ExpressionRule.getRule(expression);
913                     break;
914                 case 1:
915                     backgroundColor = (Color)aValue;
916                     break;
917                 case 2:
918                     foregroundColor = (Color)aValue;
919                     break;
920             }
921 
922             m_data.set(rowIndex, new ColorRule(expression, rule, backgroundColor, foregroundColor) );
923             // Since the value has been set, resize all of the rows(if required)
924             rulesTable.setRowHeight(ChainsawConstants.DEFAULT_ROW_HEIGHT);
925             fireTableRowsUpdated(rowIndex, rowIndex);
926         }
